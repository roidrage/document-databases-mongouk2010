# Additional metadata. #

!SLIDE javascript

    @@@ javascript
    db.posts.update(
      {name: "Notes on MongoDB"},
      {$inc: {views: 1}}
    )

!SLIDE

# Dynamic queries #

!SLIDE javascript

    @@@ javascript
    db.find({name: "Notes on MongoDB"})
